Basic process is to find differences, then do the following to print docs past the 1st
1st doc will just be the full thing
//Iterate through array of differences (remember 0 = differences between doc 1 and 2, etc)
//docNumber starts at 0 btw
for(let i = 0; i < diffArray.length; i++){
	//If it's not the last set of differences
	if(i < docs.length-1){ 
		//Iterate through the differences in a given doc
		for(let j = 0; j < diffArray[docNumber-1].length; j++){
			//If its a deletion
			if(diffArray[i][j][0] === -1)
				renderDeletionBlock(diffArray[i][j][1], docNumber)
			//or an addition
			else if(diffArray[i][j][0] === 1)
				renderAdditionBlock(diffArray[i][j][1], docNumber)
		}
	}
}

renderDeletionBlock(deletedString){
	//String from the difference in the doc to the end of the paragraph its in
	diffParagraph = docs[docNumber-1].subString(
		docs[docNumber-1].indexOf(deletedString),
		docs[docNumber-1].indexOf("\r\n\r\n", docs[docNumber-1].indexOf(deletedString)+100
		)
	diffParagraphUnchanged = diffParagraph.subString(deletedString.length)

	docSplit = docs[docNumber-1].split(deletedString)
	diffParagraphUnchangedPrior = docSplit[0].subString(
		docSplit[0].lastIndexOf("\r\n\r\n")
		)
	return diffParagraphUnchangedPrior + "<p class=\'deleted\'>" + deletedString + "</p>" + diffParagraphUnchanged
}
renderAddedBlock(addedString){
	//String from the difference in the doc to the end of the paragraph its in
	diffParagraph = docs[docNumber-1].subString(
		docs[docNumber-1].indexOf(addedString),
		docs[docNumber-1].indexOf("\r\n\r\n", docs[docNumber-1].indexOf(addedString)+100
		)
	diffParagraphUnchanged = diffParagraph.subString(addedString.length)

	docSplit = docs[docNumber-1].split(addedString)
	diffParagraphUnchangedPrior = docSplit[0].subString(
		docSplit[0].lastIndexOf("\r\n\r\n")
		)
	return diffParagraphUnchangedPrior + "<p class=\'added\'>" + addedString + "</p>" + diffParagraphUnchanged
}
